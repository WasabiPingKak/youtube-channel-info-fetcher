rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // 只允許讀取 videos/latest 這筆文件（分類資料）
    match /videos/latest {
      allow read: if true;
    }

    // 可選：允許讀取分類設定（如果有開 categories）
    match /categories/{docId} {
      allow read: if true;
    }

    // 禁止所有其他資料讀取或寫入
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
